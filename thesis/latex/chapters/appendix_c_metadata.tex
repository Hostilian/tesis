\section{C.3 Suomi-NPP VIIRS Day/Night Band (DNB)}
\begin{itemize}
  \item \textbf{Sensor Type}: Visible Infrared Imaging Radiometer Suite (VIIRS).
  \item \textbf{Spatial Resolution}: 15 arc-seconds (approximately 500 meters at the equator).
  \item \textbf{Spectral Range}: 500–900 nm (panchromatic visible/near-infrared band sensitive to low-light).
  \item \textbf{Radiometric Calibration}: Absolute radiance in units of $\text{nW}\cdot\text{cm}^{-2}\cdot\text{sr}^{-1}$.

\end{itemize}
